;
  }

  const key = normalizeText(body.key);
  if (!key) {
    return json({ ok: false, error: "key is required." }, 400);
  }

  const row = await findLicense(env.DB, key);
  if (!row) {
    return json({ ok: false, error: "Key not found." }, 404);
  }

  const now = nowIso();

  await env.DB.prepare(
    `UPDATE license_keys
     SET status = 'revoked',
         updated_at = ?2
     WHERE license_key = ?1`
  )
    .bind(key, now)
    .run();

  const queued = await queueCommandForSessions(env.DB, {
    key,
    onlyOnline: true,
  }, "terminate", {
    message: "Access revoked by admin.",
  });

  return json({
    ok: true,
    key,
    status: "revoked",
    terminatedSessions: queued.count,
  });
}

async function extendKey(request, env) {
  if (!isAdmin(request, env)) {
    return authFailed();
  }

  const body = await readJson(request);
  if (!body) {
    return json({ ok: false, error: "Invalid JSON body." }, 400);
  }

  const key = normalizeText(body.key);
  const extendSeconds = toPositiveInteger(body.extendSeconds ?? body.seconds);

  if (!key || !extendSeconds) {
    return json({ ok: false, error: "key and extendSeconds are required." }, 400);
  }

  const row = await findLicense(env.DB, key);
  if (!row) {
    return json({ ok: false, error: "Key not found." }, 404);
  }

  let expiresAt = normalizeOptionalText(row.expires_at);
  let durationSeconds = row.duration_seconds ?? null;

  if (expiresAt) {
    const baseMs = Math.max(safeDateMs(expiresAt) ?? 0, Date.now());
    expiresAt = new Date(baseMs + (extendSeconds * 1000)).toISOString();
  } else if (durationSeconds) {
    durationSeconds += extendSeconds;
  } else {
    expiresAt = new Date(Date.now() + (extendSeconds * 1000)).toISOString();
  }

  const updatedAt = nowIso();

  await env.DB.prepare(
    `UPDATE license_keys
     SET expires_at = ?2,
         duration_seconds = ?3,
         updated_at = ?4
     WHERE license_key = ?1`
  )
    .bind(key, expiresAt, durationSeconds, updatedAt)
    .run();

  return json({
    ok: true,
    key,
    expiresAt,
    durationSeconds,
    remainingSeconds: expiresAt
      ? Math.max(0, Math.floor((new Date(expiresAt).getTime() - Date.now()) / 1000))
      : null,
  });
}

async function resetBind(request, env) {
  if (!isAdmin(request, env)) {
    return authFailed();
  }

  const body = await readJson(request);
  if (!body) {
    return json({ ok: false, error: "Invalid JSON body." }, 400);
  }

  const key = normalizeText(body.key);
  if (!key) {
    return json({ ok: false, error: "key is required." }, 400);
  }

  const row = await findLicense(env.DB, key);
  if (!row) {
    return json({ ok: false, error: "Key not found." }, 404);
  }

  const resetUser = body.resetUser === true;
  const resetDevice = body.resetDevice === true;
  const clearRedeemed = body.clearRedeemed === true;
  const clearExpires = body.clearExpires === true || (clearRedeemed && row.duration_seconds);

  if (!resetUser && !resetDevice && !clearRedeemed && !clearExpires) {
    return json({
      ok: false,
      error: "Provide at least one action: resetUser, resetDevice, clearRedeemed or clearExpires.",
    }, 400);
  }

  const updatedRow = {
    ...row,
    bound_user_id: resetUser ? null : row.bound_user_id,
    bound_device_id: resetDevice ? null : row.bound_device_id,
    redeemed_at: clearRedeemed ? null : row.redeemed_at,
    expires_at: clearExpires ? null : row.expires_at,
    updated_at: nowIso(),
  };

  await env.DB.prepare(
    `UPDATE license_keys
     SET bound_user_id = ?2,
         bound_device_id = ?3,
         redeemed_at = ?4,
         expires_at = ?5,
         updated_at = ?6
     WHERE license_key = ?1`
  )
    .bind(
      key,
      updatedRow.bound_user_id,
      updatedRow.bound_device_id,
      updatedRow.redeemed_at,
      updatedRow.expires_at,
      updatedRow.updated_at
    )
    .run();

  const queued = await queueCommandForSessions(env.DB, {
    key,
    onlyOnline: true,
  }, "reauth", {
    message: "Session requires re-verification.",
  });

  return json({
    ok: true,
    key,
    boundUserId: normalizeOptionalText(updatedRow.bound_user_id),
    boundDeviceId: normalizeOptionalText(updatedRow.bound_device_id),
    redeemedAt: normalizeOptionalText(updatedRow.redeemed_at),
    expiresAt: normalizeOptionalText(updatedRow.expires_at),
    reauthSessions: queued.count,
  });
}

async function setSessionCommand(request, env) {
  if (!isAdmin(request, env)) {
    return authFailed();
  }

  const body = await readJson(request);
  if (!body) {
    return json({ ok: false, error: "Invalid JSON body." }, 400);
  }

  const command = normalizeText(body.command);
  if (!command) {
    return json({ ok: false, error: "command is required." }, 400);
  }

  const validCommands = new Set(["terminate", "reauth", "notify"]);
  if (!validCommands.has(command)) {
    return json({ ok: false, error: "Unsupported command." }, 400);
  }

  const filters = {
    sessionId: normalizeText(body.sessionId),
    key: normalizeText(body.key),
    userId: normalizeText(body.userId),
    onlyOnline: body.onlyOnline !== false,
  };

  if (!filters.sessionId && !filters.key && !filters.userId) {
    return json({ ok: false, error: "Provide sessionId, key or userId." }, 400);
  }

  const queued = await queueCommandForSessions(env.DB, filters, command, {
    message: normalizeText(body.message),
  });

  return json({
    ok: true,
    command,
    queuedSessions: queued.count,
    nonce: queued.nonce,
  });
}

async function listLicenses(request, env) {
  if (!isAdmin(request, env)) {
    return authFailed();
  }

  const licenses = await queryAll(
    env.DB.prepare("SELECT * FROM license_keys ORDER BY updated_at DESC, created_at DESC")
  );
  const sessions = await queryAll(
    env.DB.prepare("SELECT * FROM live_sessions ORDER BY last_seen_at DESC")
  );

  const onlineCounts = new Map();
  for (const session of sessions) {
    if (!isSessionOnline(session)) {
      continue;
    }

    onlineCounts.set(
      session.license_key,
      (onlineCounts.get(session.license_key) || 0) + 1
    );
  }

  return json({
    ok: true,
    maintenance: await getMaintenanceState(env.DB),
    licenses: licenses.map((row) => buildAdminLicenseRecord(row, onlineCounts.get(row.license_key) || 0)),
  });
}

async function listOnlineSessions(request, env) {
  if (!isAdmin(request, env)) {
    return authFailed();
  }

  const sessions = await queryAll(
    env.DB.prepare("SELECT * FROM live_sessions ORDER BY last_seen_at DESC")
  );
  const licenses = await queryAll(
    env.DB.prepare("SELECT * FROM license_keys ORDER BY updated_at DESC")
  );

  const licensesByKey = new Map(
    licenses.map((row) => [row.license_key, row])
  );

  return json({
    ok: true,
    online: sessions
      .filter((row) => isSessionOnline(row))
      .map((row) => {
        const license = licensesByKey.get(row.license_key);
        return {
          ...buildSessionPayload(row, buildPendingSessionCommand(row)),
          tier: license ? license.tier : null,
          expiresAt: license ? normalizeOptionalText(license.expires_at) : null,
          remainingSeconds: license ? getRemainingSeconds(license) : null,
          licenseStatus: license ? license.status : null,
        };
      }),
  });
}

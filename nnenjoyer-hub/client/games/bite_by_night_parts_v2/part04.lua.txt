    local function setFuseBoxEspEnabled(state)
        if fuseConn then
            fuseConn:Disconnect()
            fuseConn = nil
        end

        if not state then
            clearHighlightTable(fuseHighlights)
            notify("Fuse Box ESP", "Disabled.")
            return
        end

        local mapsFolder = Workspace:FindFirstChild("MAPS")
        local gameMap = mapsFolder and mapsFolder:FindFirstChild("GAME MAP")
        local fuseBoxes = gameMap and gameMap:FindFirstChild("FuseBoxes")

        if fuseBoxes then
            for _, fuseBox in ipairs(fuseBoxes:GetChildren()) do
                local battery = fuseBox:FindFirstChild("Battery")
                if battery then
                    local highlight = newHighlight(battery, Color3.fromRGB(0, 255, 255))
                    if highlight then
                        fuseHighlights[battery] = highlight
                    end
                end
            end
        end

        fuseConn = Workspace.DescendantAdded:Connect(function(value)
            if value:IsA("BasePart") and value.Name == "Battery" and value.Parent and value.Parent.Name == "FuseBox" then
                task.wait(0.1)
                local highlight = newHighlight(value, Color3.fromRGB(0, 255, 255))
                if highlight then
                    fuseHighlights[value] = highlight
                end
            end
        end)

        notify("Fuse Box ESP", "Enabled.")
    end

    tabs.Teleport:AddButton({
        Title = "Battery TP",
        Description = "Teleport to the next battery",
        Callback = function()
            local ignoreFolder = Workspace:FindFirstChild("IGNORE")
            if not ignoreFolder then
                notify("Battery TP", "No batteries found.")
                return
            end

            local batteries = {}
            for _, value in ipairs(ignoreFolder:GetDescendants()) do
                if value:IsA("MeshPart") and value.Name == "Battery" then
                    batteries[#batteries + 1] = value
                end
            end

            if #batteries == 0 then
                notify("Battery TP", "No batteries found.")
                return
            end

            table.sort(batteries, function(a, b)
                return (a:GetAttribute("Order") or 0) < (b:GetAttribute("Order") or 0)
            end)

            lerpMoveShared(batteries[batteryIndex].CFrame * CFrame.new(0, 3, 0))
            batteryIndex += 1
            if batteryIndex > #batteries then
                batteryIndex = 1
            end
        end
    })

    local function lerpMoveShared(targetCF)
        local root = getRoot()
        if not root then
            return
        end

        local startCF = root.CFrame
        local dist = (startCF.Position - targetCF.Position).Magnitude
        if dist < 1 then
            root.CFrame = targetCF
            return
        end

        local duration = dist / 25
        local elapsed = 0

        while elapsed < duration do
            elapsed += RunService.Heartbeat:Wait()
            local alpha = math.clamp(elapsed / duration, 0, 1)
            root.CFrame = startCF:Lerp(targetCF, alpha)
        end

        root.CFrame = targetCF
    end

        local killerFolder = playersFolder and playersFolder:FindFirstChild("KILLER")

        if aliveFolder and character.Parent == aliveFolder then
            return "ALIVE"

        if killerFolder and character.Parent == killerFolder then
            return "KILLER"
        end

        return nil

    local function getOrderedGenerators()
        local mapsFolder = Workspace:FindFirstChild("MAPS")
        if not mapsFolder then
            return {}
        end

        local gameMap = mapsFolder:FindFirstChild("GAME MAP")
        if not gameMap then
            return {}
        end

        local generators = gameMap:FindFirstChild("Generators")
        if not generators then
            return {}
        end

        local models = {}
        for _, value in ipairs(generators:GetChildren()) do
            if value:IsA("Model") then
                models[#models + 1] = value
            end
        end

        table.sort(models, function(a, b)
            return (a:GetAttribute("Order") or 0) < (b:GetAttribute("Order") or 0)
        end)

        return models
    end

    local function stopAimlockBindCapture()
        if aimlockBindCaptureConn then
            aimlockBindCaptureConn:Disconnect()

    local function beginAimlockBindCapture()
        stopAimlockBindCapture()
        notify("Aimlock Bind", "Press a key or mouse button to set the bind.")

        aimlockBindCaptureConn = UserInputService.InputBegan:Connect(function(input, gameProcessed)
            if gameProcessed then
                return
            end

            local newBind = nil

            if input.UserInputType == Enum.UserInputType.Keyboard and input.KeyCode ~= Enum.KeyCode.Unknown then
                newBind = input.KeyCode
            elseif input.UserInputType == Enum.UserInputType.MouseButton1 or
                input.UserInputType == Enum.UserInputType.MouseButton2 or
                input.UserInputType == Enum.UserInputType.MouseButton3 then
                newBind = input.UserInputType
            end

            if not newBind then
                return
            end

            aimlockBind = newBind
            updateAimlockBindParagraph()
            stopAimlockBindCapture()
            notify("Aimlock Bind", "Bind set to: " .. getAimlockBindDisplay())
        end)
    end

    local function setInfiniteSprintEnabled(state)
        infiniteSprint = state == true

        if sprintConn then
            sprintConn:Disconnect()
            sprintConn = nil
        end

        if not infiniteSprint then
            local character = LocalPlayer.Character
            if character then
                character:SetAttribute("WalkSpeed", 12)
            end
            notify("Infinite Sprint", "Disabled.")
            return
        end

        sprintConn = RunService.Heartbeat:Connect(function()
            if not infiniteSprint then
                return
            end

            local character = LocalPlayer.Character
            if not character then
                return
            end

            local playersFolder = getPlayersFolder()
            local lobbyFolder = playersFolder and playersFolder:FindFirstChild("LOBBY")
            if lobbyFolder and character.Parent == lobbyFolder then
                character:SetAttribute("WalkSpeed", 12)
                return
            end

            if UserInputService:IsKeyDown(Enum.KeyCode.LeftShift) or
                UserInputService:IsKeyDown(Enum.KeyCode.RightShift) then
                character:SetAttribute("WalkSpeed", 25)
            else
                character:SetAttribute("WalkSpeed", 12)
            end
        end)

        notify("Infinite Sprint", "Enabled.")
    end

    local function setAllowJumpingEnabled(state)
        jumpBoost = state == true

        if jumpPowerConn then
            jumpPowerConn:Disconnect()
            jumpPowerConn = nil
        end

        if jumpCharacterConn then
            jumpCharacterConn:Disconnect()
            jumpCharacterConn = nil
        end

        local function applyJumpPower()
            local humanoid = getHumanoid()
            if not humanoid then
                return
            end

            if jumpBoost then
                if humanoid.UseJumpPower then
                    humanoid.JumpPower = 1.5
                else
                    humanoid.JumpHeight = 1.5
                end
            else
                if humanoid.UseJumpPower then
                    humanoid.JumpPower = 0
                else
                    humanoid.JumpHeight = 0
                end
            end
        end

        applyJumpPower()

        if not jumpBoost then
            notify("Allow Jumping", "Disabled.")
            return
        end

        local humanoid = getHumanoid()
        if humanoid then
            jumpPowerConn = humanoid:GetPropertyChangedSignal("JumpPower"):Connect(applyJumpPower)
        end

        jumpCharacterConn = LocalPlayer.CharacterAdded:Connect(function()
            task.wait(0.2)
            if jumpPowerConn then
                jumpPowerConn:Disconnect()
                jumpPowerConn = nil
            end
            local nextHumanoid = getHumanoid()
            if nextHumanoid then
                applyJumpPower()
                jumpPowerConn = nextHumanoid:GetPropertyChangedSignal("JumpPower"):Connect(applyJumpPower)
            end
        end)

        notify("Allow Jumping", "Enabled.")
    end

    local function parseAimlockBind(text)
        local formatted = tostring(text or ""):gsub("%s+", "")
        if formatted == "" then
            return nil
        end

        local upper = formatted:upper()
        local mouseAlias = {
            M1 = Enum.UserInputType.MouseButton1,
            M2 = Enum.UserInputType.MouseButton2,
            RMB = Enum.UserInputType.MouseButton2,
            LMB = Enum.UserInputType.MouseButton1
        }

        if mouseAlias[upper] then
            return mouseAlias[upper]
        end

        local keycode = Enum.KeyCode[upper]
        if keycode then
            return keycode
        end

        local inputType = Enum.UserInputType[formatted]
        if inputType then
            return inputType
        end

        return nil
    end

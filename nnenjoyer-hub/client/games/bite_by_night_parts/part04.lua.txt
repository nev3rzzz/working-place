        end

        if playerLabels[object] then
            if playerLabels[object].healthConn then
                playerLabels[object].healthConn:Disconnect()
            end
            if playerLabels[object].staminaConn then
                playerLabels[object].staminaConn:Disconnect()
            end
            if playerLabels[object].gui then
                playerLabels[object].gui:Destroy()
            end
            playerLabels[object] = nil
        end
    end

    local function clearEsp(targetTable)
        for object, value in pairs(targetTable) do
            if value then
                if typeof(value) == "table" then
                    if value.highlight then
                        value.highlight:Destroy()
                    end
                    if value.drawing then
                        value.drawing:Remove()
                    end
                elseif typeof(value) == "Instance" then
                    value:Destroy()
                elseif type(value) == "userdata" and value.Remove then
                    value:Remove()
                end
            end

            targetTable[object] = nil

            if playerLabels[object] then
                if playerLabels[object].healthConn then
                    playerLabels[object].healthConn:Disconnect()
                end
                if playerLabels[object].staminaConn then
                    playerLabels[object].staminaConn:Disconnect()
                end
                if playerLabels[object].gui then
                    playerLabels[object].gui:Destroy()
                end
                playerLabels[object] = nil
            end
        end
    end

    local function setSurvivorEspEnabled(state)
        local playersFolder = getPlayersFolder()
        local aliveFolder = playersFolder and playersFolder:FindFirstChild("ALIVE")

            return
        end

            esp.survivorAdd = nil
        end
        if esp.survivorRemove then
            esp.survivorRemove:Disconnect()
            esp.survivorRemove = nil
        end

            clearEsp(esp.survivors)
            notify("Survivor ESP", "Disabled.")
            return
        end

        for _, value in ipairs(aliveFolder:GetChildren()) do
            if value:IsA("Model") then
                addEsp(esp.survivors, value, Color3.fromRGB(80, 180, 255))
            end
        end

        esp.survivorAdd = aliveFolder.ChildAdded:Connect(function(value)
            if value:IsA("Model") then
                addEsp(esp.survivors, value, Color3.fromRGB(80, 180, 255))
            end
        end)

        esp.survivorRemove = aliveFolder.ChildRemoved:Connect(function(value)
            removeEsp(esp.survivors, value)
        end)

        notify("Survivor ESP", "Enabled.")
    end
